pop_sd <- function(pop) {
  sd <- sd(pop) * sqrt((length(pop) - 1)/length(pop))
  return(sd)
} # function for converting the native sample standard deviation function to a population standard deviation
z_long <- z %>%
  pivot_longer(cols = c("height", "weight", "age", "zombies_killed", "years_of_education"), names_to = "variable", values_to = "value") # turns each variable we want to look at into one column, which can be filtered by what variable that specific row has in its "value" column using the also-new "variable" column
z_info <- z_long %>%
  group_by(variable) %>%
  summarise(mean = mean(value), standardDeviation = pop_sd(value)) # generates a table that gives a brief summary of the entire table's/population's different quantized attributes
